###############################################################
#                     INSTRUCCIONES DE EJECUCIÓN
#                 Proyecto Full Stack – (Spring / Node / React)
###############################################################

# 1. Ejecutar el Backend en Spring Boot
# -------------------------------------
# Asegúrese de estar en el directorio del backend-spring.
# Ejecutar el siguiente comando:
mvn spring-boot:run


# 2. Ejecutar el Backend en Node.js
# ---------------------------------
# Ir al directorio backend-node.
# Iniciar el servidor con:
npm start


# 3. Ejecutar el Frontend en React
# --------------------------------
# Dirígete al directorio frontend-react.
# Ejecutar el entorno de desarrollo:
npm start


###############################################################
# Una vez los servicios estén activos:
# - Spring Boot escuchará en su puerto configurado.
# - Node.js se levantará según su configuración.
# - React abrirá automáticamente la interfaz en el navegador.
###############################################################
